Form Controls 表单控件

包含各种用于表单输入的数据采集控件。

结构与样式


<!--在radio input后紧跟label，并声明label的for属性为radio的id使label与radio形成关联，label添加.ne-radio，label被点击时将联动触发radio切换；-->
<input type="radio" id="radio_1" />
<label class="ne-radio" for="radio_1"></label>

<!--勾形的radio样式：.ne-radio_hook-->
<input type="radio" id="radio_hook" />
<label class="ne-radio" for="radio_hook"></label>

<!--checkbox的实现方式同理于radio-->
<input type="checkbox" id="checkbox_1" />
<label class="ne-checkbox" for="checkbox_1"></label>

<!--利用.ne-checkbox以及使用.border-circle使其圆角边框变圆形边框便可模仿另一种风格的radio-->
<input type="radio" id="radio_circle" />
<label class="ne-checkbox border-circle" for="radio_circle"></label>

<!--同理利用.ne-switch使用checkbox实现switch开关控件-->
<input type="checkbox" id="checkbox_switch" checked="">
<label class="ne-switch" for="checkbox_switch"></label>

<!--数字输入控件numbox需声明ne-role-->
<div class="ne-numbox" ne-role="numbox">
<button class="ne-btn ne-btn-default" type="button" ne-tap="minus:numbox">-</button>
<input type="number" class="ne-input" min="1" max="20" step="2" value="1">
<button class="ne-btn ne-btn-default" type="button" ne-tap="plus:numbox">+</button>
</div>

<!--滑动选择输入控件需声明ne-role-->
<input type="range" class="ne-range" min="10" max="200" value="150" ne-role="range" />

<!--普通的文本输入控件使用样式.ne-input-->
<input class="ne-input" type="text" placeholder="placeholder..." />

<!--日期和时间输入控件需声明ne-role=“picker”，点击时会呼出选择器-->
<input class="ne-input" type="datetime" ne-role="picker" />
<input class="ne-input" type="date" ne-role="picker" />
<input class="ne-input" type="time" ne-role="picker" />

<!--多行文本输入控件.ne-textarea,设置maxlength属性便可自动计算和控制输入字数-->
<textarea class="ne-textarea" placeholder="120字，相关输入提示" rows="6" maxlength="120"></textarea>

<!--select下拉选择控件设置ne-role=“picker”也可达到选择器效果-->
<select class="ne-select" aria-label="请选择" ne-role="picker"></select>

<!--带清除按钮的文本输入控件需设置属性ne-clearable="true"-->
<input type="text" class="ne-input" ne-clearable="true">
